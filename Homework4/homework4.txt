void transpose(int *dst, int *src, int dim) {
	int i, j, k, m;
	for (i = 0; i < dim; i += 12) 
		for (j = 0; j < dim; j += 12) 
			for (k = i; (k < i + 12) & (k < dim); k++)
				for (m = j; (m < j + 12) & (m < dim); m++)
					dst[m + k * dim] = src[k + m * dim];

	for (; i < dim; i++)
		for (j = 0; j < dim; j++)
			dst[j * dim + i] = src[i * dim + j];
}
